def run_git_log(repo_path: Path) -> list[str]:
    result = subprocess.run(
        [
            "git",
            "-C",
            str(repo_path),
            "log",
            "--date=iso-strict",
            "--pretty=format:%H%x1f%an%x1f%ae%x1f%ad%x1f%s",
            "--numstat",
        ],
        check=True,
        capture_output=True,
        text=True,
    )
    return result.stdout.splitlines()


def parse_git_log(lines: list[str]) -> tuple[pd.DataFrame, pd.DataFrame]:
    commits = []
    files = []
    current = None

    for line in lines:
        if not line.strip():
            continue

        if "\x1f" in line:
            parts = line.split("\x1f")
            if len(parts) != 5:
                continue
            commit_hash, author, email, date_str, subject = parts
            current = {
                "hash": commit_hash,
                "author": author,
                "email": email,
                "date": pd.to_datetime(date_str),
                "subject": subject,
                "added": 0,
                "deleted": 0,
                "files_changed": 0,
            }
            commits.append(current)
            continue

        if current is None:
            continue

        numstat = line.split("\t")
        if len(numstat) < 3:
            continue
        added_raw, deleted_raw, path = numstat[0], numstat[1], "\t".join(numstat[2:])
        added = None if added_raw == "-" else int(added_raw)
        deleted = None if deleted_raw == "-" else int(deleted_raw)

        files.append(
            {
                "hash": current["hash"],
                "path": path,
                "added": added,
                "deleted": deleted,
            }
        )

        current["files_changed"] += 1
        if added is not None:
            current["added"] += added
        if deleted is not None:
            current["deleted"] += deleted

    commits_df = pd.DataFrame(commits)
    # Ensure date column is datetime type and convert to UTC if timezone-aware
    commits_df["date"] = pd.to_datetime(commits_df["date"], utc=True)
    commits_df = commits_df.sort_values("date")
    files_df = pd.DataFrame(files)
    return commits_df, files_df

def get_embedding(text: str, batch_size: int = 32) -> np.ndarray:
    """Generate embedding for a single text using CodeBERT."""
    # Tokenize and encode
    inputs = tokenizer(text, return_tensors='pt', truncation=True, 
                       max_length=512, padding=True).to(device)
    
    # Generate embeddings
    with torch.no_grad():
        outputs = model(**inputs)
        # Use mean pooling of last hidden state
        embedding = outputs.last_hidden_state.mean(dim=1).squeeze()
    
    return embedding.cpu().numpy()

def get_embeddings_batch(texts: list[str], batch_size: int = 8) -> np.ndarray:
    """Generate embeddings for a batch of texts."""
    embeddings = []
    
    for i in range(0, len(texts), batch_size):
        batch = texts[i:i+batch_size]
        
        # Tokenize batch
        inputs = tokenizer(batch, return_tensors='pt', truncation=True,
                          max_length=512, padding=True).to(device)
        
        # Generate embeddings
        with torch.no_grad():
            outputs = model(**inputs)
            # Mean pooling
            batch_embeddings = outputs.last_hidden_state.mean(dim=1)
            embeddings.append(batch_embeddings.cpu().numpy())
        
        if (i // batch_size + 1) % 10 == 0:
            print(f"Processed {min(i+batch_size, len(texts))}/{len(texts)} commits...")
    
    return np.vstack(embeddings)

# Train a classifier using the reference examples
# Option 1: Use similarity-based approach (compare to API examples)
# Option 2: Train a simple classifier (Logistic Regression)

print("Training classifier...")

# Method 1: Similarity-based classification
# Calculate average similarity to API examples
api_centroid = api_example_embeddings.mean(axis=0)
non_api_centroid = non_api_example_embeddings.mean(axis=0)

# Calculate similarity scores
api_similarities = cosine_similarity(commit_embeddings, api_centroid.reshape(1, -1)).flatten()
non_api_similarities = cosine_similarity(commit_embeddings, non_api_centroid.reshape(1, -1)).flatten()

# Classification based on which centroid is more similar
similarity_based_predictions = (api_similarities > non_api_similarities).astype(int)

# Method 2: Train a Logistic Regression classifier
classifier = LogisticRegression(random_state=42, max_iter=1000)
classifier.fit(X_train, y_train)

# Predict on all commits
lr_predictions = classifier.predict(commit_embeddings)
lr_probabilities = classifier.predict_proba(commit_embeddings)[:, 1]  # Probability of being API-related

# Add predictions to dataframe

# Visualize API-related commits over time
commits_df['month'] = commits_df['date'].dt.to_period("M").dt.to_timestamp()

monthly_api = commits_df.groupby('month').agg({
    'is_api_related': 'sum',
    'hash': 'count'
}).rename(columns={'hash': 'total_commits', 'is_api_related': 'api_commits'})
monthly_api['api_percentage'] = (monthly_api['api_commits'] / monthly_api['total_commits'] * 100).round(1)

fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 8))

# Plot 1: API commits vs total commits
ax1.plot(monthly_api.index, monthly_api['total_commits'], label='Total Commits', marker='o')
ax1.plot(monthly_api.index, monthly_api['api_commits'], label='API-Related Commits (CodeBERT)', marker='s', color='orange')
ax1.set_xlabel('Month')
ax1.set_ylabel('Number of Commits')
ax1.set_title('API-Related Commits Over Time (CodeBERT Classification)')
ax1.legend()
ax1.grid(True, alpha=0.3)

# Plot 2: Percentage of API commits
ax2.plot(monthly_api.index, monthly_api['api_percentage'], label='% API Commits', marker='o', color='green')
ax2.set_xlabel('Month')
ax2.set_ylabel('Percentage (%)')
ax2.set_title('Percentage of API-Related Commits')
ax2.legend()
ax2.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
